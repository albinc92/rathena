-	script	drops	-1,{
	setarray .@armor_t1[0],2301,2302,2303,2304,2305,2306;

OnNPCKillEvent:
	set .@drop, rand(1, (1000 - readparam(bLuk)));

	// Drop item
	if(.@drop < 750) {
		getmapxy(.@map$, .@x, .@y);
		makeitem .@armor_t1[rand(0, getarraysize(.@armor_t1))], 1, .@map$, rand(.@x-1, .@x+1), rand(.@y-1, .@y+1);
	}
	end;
}
